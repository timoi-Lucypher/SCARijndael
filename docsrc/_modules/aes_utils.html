[docs]def ExpandKeys(key: np.ndarray):
    '''
    Almost Numpy implementation of the ExpandKeys
    function.

    :param key: The (4, 4) array master key.
    :return: A (11, 4, 4) array reprensenting
        the AES round keys
    '''
    round_keys = [key]
    for i in range(1, 11):
        rk = []
        W = round_keys[-1]
        rk.append(SBOX[np.roll(W[-1], -1)] ^ W[0] ^ RCON[i])
        for j in range(3):
            rk.append(rk[-1] ^ W[j + 1])
        rk = np.array(rk, dtype=np.uint8)
        round_keys.append(rk)
    return np.array(round_keys, dtype=np.uint8)

[docs]def ShiftRows(states: np.ndarray):
    '''
    Numpy implementation of the ShiftRows
    function.

    :param states: (N, 4, 4) numpy array
    :return: the updated states
    '''
    s0 = states[:, :, 0].reshape(states.shape[0], 4)
    s1 = states[:, :, 1].reshape(states.shape[0], 4)
    s2 = states[:, :, 2].reshape(states.shape[0], 4)
    s3 = states[:, :, 3].reshape(states.shape[0], 4)

    r1 = np.roll(s1, -1, axis=1)
    r2 = np.roll(s2, -2, axis=1)
    r3 = np.roll(s3, -3, axis=1)

    res = np.hstack((s0, r1, r2, r3)).reshape(states.shape[0], 4, 4)
    return res.transpose((0, 2, 1))

    '''
    Numpy implementation of the MixColumns
    function. Uses galois field matrix
    multiplication using precomputed
    tables.

    :param states: (N, 4, 4) numpy array
    :return: the updated states
    '''
    r = states
    res = np.zeros(states.shape, dtype=np.uint8)

    # 2 * a0 + a3 + a2 + 3 * a1
    res[:,:,0] = GF_MULT2[r[:,:,0]] ^ r[:,:,3] ^ r[:,:,2] ^ GF_MULT3[r[:,:,1]]
    # 2 * a1 + a0 + a3 + 3 * a2
    res[:,:,1] = GF_MULT2[r[:,:,1]] ^ r[:,:,0] ^ r[:,:,3] ^ GF_MULT3[r[:,:,2]]
    # 2 * a2 + a1 + a0 + 3 * a3
    res[:,:,2] = GF_MULT2[r[:,:,2]] ^ r[:,:,1] ^ r[:,:,0] ^ GF_MULT3[r[:,:,3]]
    #2 * a3 + a2 + a1 + 3 * a0 
    res[:,:,3] = GF_MULT2[r[:,:,3]] ^ r[:,:,2] ^ r[:,:,1] ^ GF_MULT3[r[:,:,0]]
    return res
